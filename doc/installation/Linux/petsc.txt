------------------------------------------------------------------------------------------
PREREQUISITES
------------------------------------------------------------------------------------------

sudo apt-get update
sudo apt-get install build-essential
sudo apt-get install gfortran
sudo apt-get install libmpich2-dev
sudo apt-get install cmake
sudo apt-get install git-all

------------------------------------------------------------------------------------------
PETSC DEBUG
------------------------------------------------------------------------------------------

Download and unpack PETSC version 3.5.4

cd petsc-3.5.4

./configure \
--prefix=/opt/petsc/petsc-3.5.4-int32-debug \
--download-fblaslapack=1 \
--with-debugging=1 \
--COPTFLAGS="-g -O0" \
--FOPTFLAGS="-g -O0" \
--CXXOPTFLAGS="-g -O0" \
--with-large-file-io=1 \
--with-shared-libraries=0 \
--with-cc=mpicc \
--with-cxx=mpicxx \
--with-fc=mpif90 \
--download-metis=1 \
--download-parmetis=1 \
--download-scalapack=1 \
--download-mumps=1 \
--download-superlu_dist=1 \
--with-clean

make all

make test

sudo make install

------------------------------------------------------------------------------------------

------------------------------------------------------------------------------------------
OpenBLAS
------------------------------------------------------------------------------------------

Download & unpack (http://www.openblas.net/)

make BINARY=64 USE_THREAD=0 CC=gcc FC=gfortran

sudo make PREFIX=/opt/openblas install

------------------------------------------------------------------------------------------
PETSC OPTIMIZED
------------------------------------------------------------------------------------------

./configure \
--prefix=/opt/petsc/petsc-3.5.4-int32-opt \
--with-blas-lapack-lib=/opt/openblas/lib/libopenblas.a \
--with-debugging=0 \
--COPTFLAGS="-O3" \
--FOPTFLAGS="-O3" \
--CXXOPTFLAGS="-O3" \
--with-large-file-io=1 \
--with-shared-libraries=0 \
